  8 /** @namespace
  9  *  
 10  *  @param config {Object} Central object that defines the application models, views, controllers. The external API can also be defined by setting a property of exports.
 11  *
 12  */
 13 var mvc = function(config) {
 14     
 15     var events, dependencies;
 16 
 17     events = mvc.events();
 18     dependencies = mvc.dependencies();
 19 
 20     if(!config)
 21       throw new Error('No config object found');
 22 
 23     if(config.imports)
 24       mvc.imports(config.import, mvc, dependencies);
 25     
 26     if(config.models)
 27       mvc.models(config.models, events, dependencies);
 28 
 29     if(config.views)
 30       mvc.views(config.views, events, dependencies);
 31 
 32     if(config.controllers)
 33       mvc.controllers(config.controllers, events, dependencies);
 34 
 35     if(config.exports)
 36       return mvc.exports(config.exports, events);
 37 }
 38 
 39 /** @namespace 
 40  *  
 41  *  @param models {Object} Collection of model objects. Each must define a facade, which is given a reference to events.dispatch and is registered as a dependency object.
 42  *  @param events {Object} Reference to the events object.
 43  *  @param dependencies {Object} Reference to the dependencies object
 44  *
 45  */
 46 mvc.models = function(models, events, dependencies) {
 47     if(!models)
 48       throw new Error('No models found');
 49     
 50     _.each(models, function(model, key) {
 51         dependencies.register(key, model.facade);
 52 
 53         model.facade.dispatch = events.dispatch;
 54 
 55         if(model.facade.init)
 56           model.facade.init()
 57     });
 58 }
 59 
 60 /** @namespace 
 61  *
 62  *  @param views {Object} Collection of views. Each must define a mediator object, which is given a reference to events.dispatch. If an init method is found it will be automatically called, any other methods are registered as event listeners. Optionally a view element and external dependencies can be defined.
 63  *  @param events {Object} Reference to the the object.
 64  *  @param dependencies {Object} Reference to the dependencies object
 65  */
 66 mvc.views = function(views, events, dependencies) {
 67     var setup_mediator, register_listeners, check_views;
 68 
 69     /** @private */
 70     check_views = function(views) {
 71         if(!views)
 72           throw new Error('A view object must be passed');
 73     }
 74 
 75     /** @private */
 76     setup_mediator = function(view) {
 77         if(!view.mediator)
 78           throw new Error('A mediator must be defined on view objects');
 79 
 80         if(view.element)
 81           view.mediator.element = view.element;
 82 
 83         if(view.requires)
 84           dependencies.inject(view.mediator, view.requires);
 85         
 86         view.mediator.dispatch = events.dispatch;
 87 
 88         if(view.mediator.init)
 89           view.mediator.init();
 90     }
 91 
 92     /** @private */
 93     register_listeners = function(view) {
 94         _.each(_.functions(view.mediator), function(method) {
 95             if(method !== 'init' && method !== 'dispatch')
 96               events.listen(method, view.mediator[method], view.mediator);
 97         });
 98     }
 99 
100     check_views(views);
101 
102     _.each(views, function(view) {
103         register_listeners(view);
104         setup_mediator(view);
105     });
106 }

108 /** @namespace
109  *
110  *  @param controllers {Object} Collection of functions which are mapped to defined events.
111  *  @param events {Object} Reference to the the object.
112  *  @param dependencies {Object} Reference to the dependencies object
113  *
114  */
115 mvc.controllers = function(controllers, events, dependencies) {
116     
117     _.each(controllers, function(controller, event) {
118         var context = {
119             dispatch: events.dispatch
120         }
121 
122         if(!_.isFunction(controller.command))
123           throw new Error('No command function found');
124 
125         if(controller.requires && !_.isArray(controller.requires))
126           throw new Error('Requirements must be an array');
127 
128         if(controller.requires)
129           dependencies.inject(context, controller.requires);
130 
131         events.listen(event, controller.command, context);
132     });
133 }

135 /** @namespace
136  *  
137  *  @param api {Object} External API for controlling the application
138  *  @param events {Object} Reference to the events object
139  *
140  */
141 mvc.exports = function(api, events) {
142 
143     var exports, context;
144 
145     exports = {},
146         
147     context = {
148         dispatch: events.dispatch,
149         listen: events.listen
150     }
151 
152     _.each(api, function(method, key) {
153         
154         exports[key] = function() {
155             method.apply(context, arguments);
156         }
157 
158     });
159 
160     return exports;
161 }

163 /** @namespace
164  *  
165  *  @param requires {Object} Applications that the parent application relies upon.
166  *  @param init {Object} Reference to the main mvc function
167  *
168  */
169 mvc.imports = function(modules, init, dependencies) {
170 
171     if(!_.isFunction(init))
172       throw new Error('no init function found');
173 
174     if(!dependencies)
175       throw new Error('no dependency object found');
176 
177     _.each(modules, function(module, name) {
178         var instance = init(module);
179 
180         dependencies.register(name, instance);
181     });
182 
183 }

185 /** @namespace */
186 mvc.events = function() {
187 
188     var registered, register_event, check_event, check_params, check_callback;
189 
190     /** @private */
191     registered = {};
192 
193     /** @private */
194     register_event = function(event, callback, context) {
195         check_event(event);
196         check_callback(callback);
197         
198         if(!registered[event])
199           registered[event] = [];
200 
201         registered[event].push({ fn: callback, context: context });
202     }
203 
204     /** @private */
205     check_event = function(event) {
206         if(typeof event !== 'string')
207           throw new Error('An event string must be passed to events.dispatch');
208     }
209 
210     /** @private */
211     check_params = function(params) {
212         if(params && !_.isArray(params))
213           throw new Error('params must be an array');
214     }
215 
216     /** @private */
217     check_callback = function(callback) {
218         if(!_.isFunction(callback))
219           throw new Error('A callback function must be passed to events.dispatch');
220     }
221 
222     return {
223 
224         /** @scope mvc.events
225          *
226          *  @param event {String} Unique identifier that triggers all registered callbacks
227          *  @param param {Array} Optional arguments to pass to the callback functions
228          *
229          */
230         dispatch: function(event, params) {
231             check_event(event);
232             check_params(params);    
233         
234             if(registered[event]) {
235                 _.each(registered[event], function(callback) {
236                     var context = callback.context || this;
237 
238                     callback.fn.apply(context, params);
239                 });
240             }
241         },
242 
243         /**
244          *
245          * @param event {String} Unique identifier that selects the callbacks to be triggered
246          * @param callback {Function} Called when event string is dispatched
247          * @param context {Object} Context in whcih the callback is applied. By default the context is the event object
248          *
249          */
250         listen: function(event, callback, context) {
251             register_event(event, callback, context);
252         }
253 
254     }
255     
256 };

258 /** @namespace */
259 mvc.dependencies = function() {
260 
261     var registered, check_registered, check_dependency, check_target, check_requires;
262 
263     /** @private */
264     registered = {};
265 
266     /** @private */
267     check_registered = function(name) {
268         if(!_.isString(name))
269           throw new Error('Name parameter must be a string');
270 
271         if(registered[name])
272           throw new Error('Dependency already exists');
273     }
274 
275     /** @private */
276     check_dependency = function(dependency, name) {
277         if(!dependency)
278           throw new Error('No dependency object found for ' + name);
279     }
280 
281     /** @private */
282     check_target = function(target) {
283         if(!target)
284           throw new Error('No target defined');
285     }
286 
287     /** @private */
288     check_requires = function(requires) {
289         if(!_.isArray(requires))
290           throw new Error('No requires array found');
291     }
292    
293     return {
294 
295         /**
296          *
297          *  @param target {Object} Target object that dependencies are inject into
298          *  @param requires {Array} List of strings to identify what the target object depends upon
299          *
300          */
301         inject: function(target, requires) {
302             check_target(target);
303             check_requires(requires);
304 
305             _.each(requires, function(dependency) {
306                 target[dependency] = registered[dependency];
307             });
308         },
309 
310         /**
311          *
312          *  @param name {String} Used to pull back the object as a dependency
313          *  @param dependency {Object} Any predefined dependencies will be injected if found
314          *
315          */
316         register: function(name, dependency) {
317             check_dependency(dependency, name);
318             check_registered(name);
319 
320             registered[name] = dependency;
321         }
322     }
323 
324 }
