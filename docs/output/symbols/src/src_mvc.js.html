  8 /** @namespace
  9  *
 10  *  Used to create instances of mvc.js applications. It takes a config object that defines the application objects.
 11  *  
 12  *  @param config {Object} Used to define the application's models, views, and controllers as well as arbitrary values. It also provides the mechanism for importing ubmodules using the imports object and defining a return API by defining an exports object.
 13  *  
 14  */
 15 var mvc = function(config) {
 16     
 17     var events, dependencies, models, views, controllers, values, imports, exports;
 18 
 19     events = mvc.events();
 20     dependencies = mvc.dependencies();
 21     values = mvc.values(dependencies);
 22     models = mvc.models(events, dependencies);
 23     views = mvc.views(events, dependencies);
 24     controllers = mvc.controllers(events, dependencies);
 25     exports = mvc.exports(events);
 26     imports = mvc.imports(mvc, dependencies);
 27 
 28     if(!config)
 29       throw new Error('No config object found');
 30 
 31     if(config.imports)
 32       imports(config.imports);
 33 
 34     if(config.values)
 35       values(config.values);
 36     
 37     if(config.models)
 38       models(config.models);
 39 
 40     if(config.views)
 41       views(config.views);
 42 
 43     if(config.controllers)
 44       controllers(config.controllers);
 45 
 46     if(config.exports)
 47       return exports(config.exports);
 48 
 49 }

 51 /** @namespace 
 52  *
 53  *  Takes a collection of models objects and defines each as a dependency using the object's key as the id. Each model is given a reference to the events.dispatch method to allow the system to be notified when updates are made.
 54  *  
 55  *  @param models {Object} Collection of model objects. Each must contain a facade object which defines a public API. Optionally dependencies can be set by defining a requires property with an array of strings.
 56  *  @param events {Object} Reference to the events object for the current application.
 57  *  @param dependencies {Object} Reference to the dependencies object for the current application.
 58  *
 59  */
 60 mvc.models = function(events, dependencies) {
 61 
 62     var register;
 63     
 64     register = function(models) {
 65         if(!models)
 66           throw new Error('No models found');
 67 
 68         _.each(models, function(model, key) {
 69             if(!model.facade)
 70               throw new Error('No facade found on ' + key + ' model');
 71 
 72             dependencies.register(key, model.facade);
 73 
 74             model.facade.dispatch = events.dispatch;
 75 
 76             if(model.requires)
 77               dependencies.inject(model.facade, model.requires);
 78 
 79             if(model.facade.init)
 80               model.facade.init()
 81         });
 82     }
 83 
 84     return register;
 85 
 86 }

 88 /** @namespace 
 89  *
 90  *  Takes a collection of views and assigns references to dispatch, listen, the view element, any required dependencies.
 91  *
 92  *  @param views {Object} Collection of views. Each must define a mediator object, which is given a reference to events.dispatch and events.listen. If an init method is found it will be automatically called, any other public methods are automatically registered as event listeners. Optionally a view element can be defined as well as any external dependencies by setting an array of strings on the requires property.
 93  *  @param events {Object} Reference to the the events object for the current application.
 94  *  @param dependencies {Object} Reference to the dependencies object for the current application.
 95  */
 96 mvc.views = function(events, dependencies) {
 97 
 98     var setup_mediator, register_listeners, register;
 99 
100     /** @private */
101     setup_mediator = function(view, name) {
102 
103         if(!view.mediator)
104           throw new Error('No mediator object found for ' + name + ' view');
105 
106         if(view.element)
107           view.mediator.element = view.element;
108 
109         if(view.requires)
110           dependencies.inject(view.mediator, view.requires);
111         
112         view.mediator.dispatch = events.dispatch;
113 
114         if(view.mediator.init)
115           view.mediator.init();
116 
117     }
118 
119     /** @private */
120     register_listeners = function(view) {
121 
122         _.each(_.functions(view.mediator), function(method) {
123             if(method !== 'init' && method !== 'dispatch')
124               events.listen(method, view.mediator[method], view.mediator);
125         });
126 
127     }
128 
129     /** @private */
130     register = function(views) {
131         if(!views)
132           throw new Error('A view object must be passed');
133 
134         _.each(views, function(view, name) {
135             register_listeners(view, name);
136             setup_mediator(view);
137         });
138     }
139 
140     return register;
141 
142 }

144 /** @namespace
145  *
146  *  Takes a collection of conntroller objects, which must define a command function. The command function is provided with references to dispatch, listen and any dependencies listed in the requires array.
147  *
148  *  @param controllers {Object} Collection of functions which are mapped to defined events.
149  *  @param events {Object} Reference to the the events object for the current application.
150  *  @param dependencies {Object} Reference to the dependencies object for the current application.
151  *
152  */
153 mvc.controllers = function(events, dependencies) {
154 
155     var context, register;
156     
157     context = {
158         dispatch: events.dispatch
159     }
160 
161     register = function(controllers) {
162         _.each(controllers, function(controller, event) {
163             if(!_.isFunction(controller.command))
164               throw new Error('No command function found on ' + event + ' controller');
165 
166             if(controller.requires && !_.isArray(controller.requires))
167               throw new Error('requires property for ' + event + ' controller must be an array of strings');
168 
169             if(controller.requires)
170               dependencies.inject(context, controller.requires);
171 
172             events.listen(event, controller.command, context);
173         });
174     }
175 
176     return register;
177 
178 }

180 /** @namespace
181  *
182  *  Maps a collection of arbitrary values as dependencies. Can be used to set application wide configuration properties.
183  *  
184  *  @param values {Object} Object of values to register as available dependencies
185  *  @param events {Object} Reference to the dependencies object for the current application.
186  *
187  */
188 mvc.values = function(dependencies) {
189 
190     var register;
191 
192     register = function(values) {
193         if(!values)
194           throw new Error('No values defined');
195 
196         _.each(values, function(value, key) {
197             dependencies.register(key, value);
198         });
199     }
200 
201     return register;
202 
203 }

205 /** @namespace
206  *
207  */
208 mvc.plugins = function(events, dependencies) {
209 
210     var registered, context;
211 
212     registered = {}
213 
214     context = {
215         dispatch: events.dispatch,
216         listen: events.listen,
217         dependencies: dependencies
218     };
219 
220     return {
221 
222         register: function(plugins) {
223             _.extend(registered, plugins);
224         },
225 
226         apply: function(config) {
227             _.each(config, function(item, key) {
228                 if(registered[key])
229                   registered[key].call(context, item);
230             });
231         }
232 
233     };
234 
235 }

237 /** @namespace
238  *
239  *  Defines that object that will be returned when the mvc function is called. Each function has access to listen and dispatch.
240  *  
241  *  @param api {Object} Collection of functions that will be externally available.
242  *  @param events {Object} Reference to the events objec for the current application.
243  *
244  */
245 mvc.exports = function(events) {
246 
247     var exports, context, register;
248 
249     exports = {};
250         
251     context = {
252         dispatch: events.dispatch,
253         listen: events.listen
254     }
255 
256     register = function(api) {
257 
258         _.each(api, function(method, key) {
259             
260             exports[key] = function() {
261                 method.apply(context, arguments);
262             }
263 
264         });
265 
266         return exports;
267 
268     }
269 
270     return register;
271 
272 }

274 /** @namespace
275  *
276  *  Takes a collection of application config objects and creates a new instance of each. The imported applications are then registered as dependencies on the current application using the key string as the id.
277  *  
278  *  @param modules {Object} Applications that the parent application relies upon. Each object is automatically used to create an mvc application.
279  *  @param init {Object} Reference to the main mvc function for the current application.
280  *  @param dependencies {Object} Reference to the dependencies object for the current application.
281  *
282  */
283 mvc.imports = function(init, dependencies) {
284 
285     var register;
286 
287     if(!_.isFunction(init))
288       throw new Error('no init function found');
289 
290     if(!dependencies)
291       throw new Error('no dependency object found');
292 
293     register = function(modules) {
294         _.each(modules, function(module, name) {
295             var instance = init(module);
296 
297             dependencies.register(name, instance);
298         });
299     }
300 
301     return register;
302 
303 }

305 /** @namespace
306  *
307  *  Implements the observer pattern by returning an object with dispatch and listen methods.
308  *
309  */
310 mvc.events = function() {
311 
312     var registered, register_event, check_event, check_params, check_callback;
313 
314     /** @private */
315     registered = {};
316 
317     /** @private */
318     register_event = function(event, callback, context) {
319         check_event(event);
320         check_callback(callback);
321         
322         if(!registered[event])
323           registered[event] = [];
324 
325         registered[event].push({ fn: callback, context: context });
326     }
327 
328     /** @private */
329     check_event = function(event) {
330         if(typeof event !== 'string')
331           throw new Error(event + ' should be a string.');
332     }
333 
334     /** @private */
335     check_params = function(params) {
336         if(params && !_.isArray(params))
337           throw new Error(params + ' should be an array.');
338     }
339 
340     /** @private */
341     check_callback = function(callback) {
342         if(!_.isFunction(callback))
343           throw new Error(callback + ' should be a function.');
344     }
345 
346     return {
347 
348         /** 
349          *
350          *  Calls any functions listening for the event string.
351          *
352          *  @param event {String} Unique identifier that triggers all registered callbacks
353          *  @param param {Array} Optional arguments to pass to the callback functions
354          *
355          */
356         dispatch: function(event, params) {
357             check_event(event);
358             check_params(params, event);    
359         
360             if(registered[event]) {
361                 _.each(registered[event], function(callback) {
362                     var context = callback.context || this;
363 
364                     callback.fn.apply(context, params);
365                 });
366             }
367         },
368 
369         /**
370          *
371          *  Registers the callback function as a listener for the defined event string.
372          *
373          *  @param event {String} Unique identifier that selects the callbacks to be triggered
374          *  @param callback {Function} Called when event string is dispatched
375          *  @param context {Object} Context in whcih the callback is applied. By default the context is the event object
376          *
377          */
378         listen: function(event, callback, context) {
379             register_event(event, callback, context);
380         }
381 
382     }
383     
384 };

386 /** @namespace
387  *
388  *  Returns an object with inject and register methods.
389  *
390  * */
391 mvc.dependencies = function() {
392 
393     var registered, check_registered, check_dependency, check_target, check_requires;
394 
395     /** @private */
396     registered = {};
397 
398     /** @private */
399     check_registered = function(name) {
400         if(!_.isString(name))
401           throw new Error(name + ' should be a string');
402 
403         if(registered[name])
404           throw new Error(name + ' already exists');
405     }
406 
407     /** @private */
408     check_dependency = function(dependency, name) {
409         if(!dependency)
410           throw new Error(name + ' does not exist as a registered dependency.');
411     }
412 
413     /** @private */
414     check_target = function(target) {
415         if(!target)
416           throw new Error('No target defined');
417     }
418 
419     /** @private */
420     check_requires = function(requires) {
421         if(!_.isArray(requires))
422           throw new Error(requires + ' should be an array');
423     }
424    
425     return {
426 
427         /**
428          *  
429          *  Appends the required dependencies as properties on the target object.
430          *
431          *  @param target {Object} Target object that dependencies are inject into
432          *  @param requires {Array} List of strings to identify what the target object depends upon
433          *
434          */
435         inject: function(target, requires) {
436             check_target(target);
437             check_requires(requires);
438 
439             _.each(requires, function(dependency) {
440                 target[dependency] = registered[dependency];
441             });
442         },
443 
444         /**
445          *
446          *  Registers objects as dependencies that can be injected into other objects.
447          *
448          *  @param name {String} Used to pull back the object as a dependency
449          *  @param dependency {Object} Any predefined dependencies will be injected if found
450          *
451          */
452         register: function(name, dependency) {
453             check_dependency(dependency, name);
454             check_registered(name);
455 
456             registered[name] = dependency;
457         }
458     }
